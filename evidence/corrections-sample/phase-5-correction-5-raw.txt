**Real-Time Subscription Authorization Enforcement:**

Every SSE connection enforces role-based authorization:

```javascript
// Backend: GET /api/v1/events (SSE endpoint)

async function handleEventsStream(req, res) {
  // 1. Verify JWT token from request header
  const token = req.headers.authorization?.split('Bearer ')[1];
  const user = await verifyJWT(token);
  
  if (!user) {
    res.status(401).json({ error: 'Unauthorized' });
    return;
  }
  
  // 2. Extract operator_id and role from JWT claims
  const { operator_id, role } = user;
  
  // 3. Set up SSE response headers
  res.setHeader('Content-Type', 'text/event-stream');
  res.setHeader('Cache-Control', 'no-cache');
  res.setHeader('Connection', 'keep-alive');
  
  // 4. Subscribe to events with authorization filter
  const eventEmitter = getEventBus();
  
  const handleEvent = (event) => {
    // Only emit events that user is authorized to see
    const { booking_id, resource_operator_id, event_severity } = event;
    
    // Authorization check 1: User only sees events for their operator
    if (resource_operator_id !== operator_id) {
      return; // Silently skip event (user not authorized)
    }
    
    // Authorization check 2: Operations role does not see financial events
    if (role === 'operations' && event.type.includes('refund')) {
      return; // Skip financial events for operations role
    }
    
    // Authorization check 3: Critical issues only sent to owner/operations
    if (event_severity === 'critical' && role === 'support') {
      return; // Skip critical escalations for support role
    }
    
    // Event passed all authorization checks; emit to client
    res.write(`event: ${event.type}\n`);
    res.write(`data: ${JSON.stringify(event)}\n\n`);
  };
  
  eventEmitter.on('event', handleEvent);
  
  // 5. Handle connection close
  req.on('close', () => {
    eventEmitter.off('event', handleEvent);
    res.end();
  });
  
  // 6. Heartbeat to keep connection alive and detect disconnects
  const heartbeat = setInterval(() => {
    res.write(': heartbeat\n\n');
  }, 30000); // Every 30 seconds
  
  req.on('close', () => clearInterval(heartbeat));
}
```

**What this prevents:**

- **Cross-operator data leakage:** Lucia cannot see Rafael Mendoza's (different operator) booking events by subscribing to SSE
- **Role-based filtering:** Operations role does not receive financial/refund events in real-time
- **Unauthorized escalations:** Support role does not receive "critical issue" escalations (only Lucia/Rafael)
- **Polling optimization:** Frontend knows what events to expect, can batch cache invalidations accordingly

### 6.2.5 File Upload Security Pipeline

**File upload endpoints and security constraints:**

**No file uploads in MVP.** Itineraries are generated server-side (HTML → PDF via Puppeteer), not user-uploaded. However, future features may require file uploads (e.g., client documents, course photos). Security pipeline defined for future use:

**Constraint Matrix (Future File Upload Features):**

| Feature | Max File Size | Allowed MIME Types | Server-Side Validation | Storage Location | Scanning Required |
|---------|---|---|---|---|---|
| Client document upload (future) | 10 MB | PDF, DOCX, JPG, PNG | Verify MIME type, scan for viruses | Supabase Storage (private) | ClamAV scan |
| Course photo (admin only) | 5 MB | JPG, PNG, WebP | Verify dimensions (min 1200x800), strip EXIF | Supabase Storage (public) | No |
| Itinerary custom branding (future) | 2 MB | SVG, PNG | SVG sanitization (remove scripts) | Supabase Storage (private) | Yes (XSS) |

**File Upload Security Implementation:**

```typescript
// src/lib/file-upload.ts
import { createClient } from '@supabase/supabase-js';
import sharp from 'sharp'; // Image processing
import DOMPurify from 'isomorphic-dompurify'; // SVG sanitization

const UPLOAD_LIMITS = {
  client_document: { maxSize: 10 * 1024 * 1024, allowedMimes: ['application/pdf', 'application/vnd.openxmlformats-officedocument.wordprocessingml.document', 'image/jpeg', 'image/png'] },
  course_photo: { maxSize: 5 * 1024 * 1024, allowedMimes: ['image/jpeg', 'image/png', 'image/webp'] },
  branding_logo: { maxSize: 2 * 1024 * 1024, allowedMimes: ['image/svg+xml', 'image/png'] }
};

async function validateAndUploadFile(
  file: File,
  uploadType: 'client_document' | 'course_photo' | 'branding_logo',
  resourceId: string
): Promise<{ path: string; error?: string }> {
  const limits = UPLOAD_LIMITS[uploadType];
  
  // 1. Size validation
  if (file.size > limits.maxSize) {
    return { error: `File exceeds ${limits.maxSize / 1024 / 1024}MB limit` };
  }
  
  // 2. MIME type validation (client + server)
  if (!limits.allowedMimes.includes(file.type)) {
    return { error: `File type ${file.type} not allowed. Allowed: ${limits.allowedMimes.join(', ')}` };
  }
  
  // 3. Server-side MIME type verification (re-verify to prevent spoofing)
  const buffer = await file.arrayBuffer();
  const magic = new Uint8Array(buffer.slice(0, 12));
  
  // Magic byte checks (sample)
  const isPDF = magic[0] === 0x25 && magic[1] === 0x50 && magic[2] === 0x44; // %PDF
  const isJPEG = magic[0] === 0xFF && magic[1] === 0xD8 && magic[2] === 0xFF;
  const isPNG = magic[0] === 0x89 && magic[1] === 0x50 && magic[2] === 0x4E; // \x89PNG
  
  if (file.type === 'application/pdf' && !isPDF) {
    return { error: 'Invalid PDF file (failed magic byte check)' };
  }
  
  // 4. Image-specific processing (strip EXIF, re-encode)
  if (uploadType === 'course_photo' && file.type.startsWith('image/')) {
    try {
      const metadata = await sharp(buffer).metadata();
      
      // Dimension validation
      if ((metadata.width || 0) < 1200 || (metadata.height || 0) < 800) {
        return { error: 'Image must be at least 1200x800 pixels' };
      }
      
      // Strip EXIF and re-encode (removes metadata that could leak location)
      const optimizedBuffer = await sharp(buffer)
        .withMetadata(false) // Strip EXIF
        .toBuffer();
      
      const supabase = createClient(process.env.SUPABASE_URL, process.env.SUPABASE_KEY);
      const { data, error } = await supabase.storage
        .from('course-photos')
        .upload(`${resourceId}/${file.name}`, optimizedBuffer, { 
          contentType: file.type,
          upsert: true 
        });
      
      return { path: data?.path || '', error: error?.message };
    } catch (err) {
      return { error: 'Failed to process image' };
    }
  }
  
  // 5. SVG sanitization (remove scripts, event handlers)
  if (uploadType === 'branding_logo' && file.type === 'image/svg+xml') {
    const text = await file.text();
    const sanitized = DOMPurify.sanitize(text, { 
      ALLOWED_TAGS: ['svg', 'path', 'circle', 'rect', 'line', 'text', 'g'], 
      ALLOWED_ATTR: ['d', 'cx', 'cy', 'r', 'x', 'y', 'width', 'height', 'viewBox', 'fill', 'stroke'] 
    });
    
    const supabase = createClient(process.env.SUPABASE_URL, process.env.SUPABASE_KEY);
    const { data, error } = await supabase.storage
      .from('branding')
      .upload(`${resourceId}/logo.svg`, sanitized, { contentType: 'image/svg+xml', upsert: true });
    
    return { path: data?.path || '', error: error?.message };
  }
  
  // 6. Standard upload for other types
  const supabase = createClient(process.env.SUPABASE_URL, process.env.SUPABASE_KEY);
  const { data, error } = await supabase.storage
    .from(uploadType.replace('_', '-'))
    .upload(`${resourceId}/${file.name}`, buffer, { contentType: file.type });
  
  return { path: data?.path || '', error: error?.message };
}

// Usage in API endpoint
app.post('/api/v1/client-documents/:bookingId', async (req, res) => {
  const { file } = req.body; // multipart form data
  const result = await validateAndUploadFile(file, 'client_document', req.params.bookingId);
  
  if (result.error) {
    res.status(400).json({ error: result.error });
  } else {
    res.json({ data: { path: result.path } });
  }
});
```

**Rate Limit State Storage (Distributed Deployments):**

For MVP (single Railway instance), in-memory rate limit state is sufficient. However, as platform scales or multiple instances are deployed, rate limit state must be shared:

**Single-Server (Day Zero → Month 3):**
```javascript
// In-memory store (resets on restart)
const rateLimitStore = new Map<string, { count: number; resetAt: number }>();

app.use((req, res, next) => {
  const userId = req.user?.id || req.ip;
  const now = Date.now();
  
  const limit = rateLimitStore.get(userId) || { count: 0, resetAt: now + 60000 };
  
  if (now > limit.resetAt) {
    limit.count = 1;
    limit.resetAt = now + 60000;
  } else {
    limit.count++;
  }
  
  rateLimitStore.set(userId, limit);
  
  if (limit.count > 1000) {
    return res.status(429).json({ error: 'RATE_LIMITED' });
  }
  
  res.setHeader('X-RateLimit-Remaining', 1000 - limit.count);
  next();
});
```

  const key = `ratelimit:${userId}`;

  if (current === 1) {
    // First request in window, set expiration
    await redis.expire(key, 60); // 60 second window

**Per-Table Concurrency Control Strategy:**

For every table where concurrent edits are possible, specify conflict resolution approach:

| Table | Concurrent Edit Scenario | Strategy | Implementation |
|-------|--------------------------|----------|-----------------|
| `bookings` | Lucia reschedules dates while Rafael approves course change | **Optimistic Locking** (version field) | Add `version INT DEFAULT 0` field; increment on each update; reject if version mismatch |
| `payments` | Stripe webhook arrives while admin is issuing refund | **Pessimistic Locking** (row lock during critical section) | `SELECT * FROM payments FOR UPDATE` during refund check; locks row until transaction completes |
| `refunds` | Two admins (future Finance Manager + Rafael) approve same refund | **Last-Write-Wins (safe)** | Refund can only be approved once (status field prevents re-approval); second attempt fails gracefully |
| `course_requests` | Course contact clicks "confirm" while Lucia marks as "denied" | **Pessimistic Locking** (state machine enforcement) | Only certain state transitions allowed (pending → confirmed or pending → denied); race condition impossible |
| `issues` | Lucia logs issue while Rafael's background job processes same booking | **Event Sourcing** (log all changes) | Every issue change is logged as immutable event; no in-place updates; avoids conflicts |
| `tier2_approvals` | Lucia submits request while Rafael reviews same request | **Optimistic Locking + User Notification** | If request already decided, notify requester instead of allowing duplicate |
| `clients` | Two team members edit same client record simultaneously | **Last-Write-Wins** (acceptable for reference data) | Client data is not critical; latest update wins; audit log tracks all changes |

**Implementation Example (Optimistic Locking for Bookings):**

  // ... existing fields ...
  version: integer('version').default(0).notNull(),
});

// Update function with optimistic locking
async function updateBookingDates(bookingId: string, newCheckInDate: Date, expectedVersion: number) {
  const result = await db
    .update(bookings)
    .set({
      check_in_date: newCheckInDate,
      version: sql`${bookings.version} + 1`, // Increment version
      updated_at: new Date(),
    })
    .where(
      and(
        eq(bookings.id, bookingId),
        eq(bookings.version, expectedVersion) // Ensure no concurrent changes
      )
    );
  
  if (result.rowCount === 0) {
    // No rows updated; version mismatch (concurrent edit detected)
    throw new Error('Booking was modified by another user. Please refresh and try again.');
  }
  
  return result;
}

- Decision: Use Neon PostgreSQL, not Supabase PostgreSQL or AWS RDS

- Decision: Use Clerk, not Supabase Auth or Auth0
- Rationale: Standalone service (not bundled with database), 10K free MAU, SSO-ready, developer-friendly
- Trade-offs: More expensive than Supabase Auth at scale (but Supabase Auth has reliability issues)
- Research: Perplexity Sonar Pro (pricing, Feb 2026), Grok sentiment analysis (Supabase Auth deteriorated Feb 2026 due to India IT Act outages)

- Rationale: Three-layer enforcement (database RLS, API middleware, UI authorization); single layer failure doesn't break security
- Trade-offs: Complexity (requires RLS policy definition + API layer checks); learning curve; cannot be bypassed
- Research: Security best practices (defense in depth); founder requirement ("hard guardrails")
- Status: LOCKED IN
- Cost Impact: $0 (built into PostgreSQL)

**D-55-P5-012: Refund Workflow Buffer (24 Hours)**
- Decision: 24-hour approval buffer before refund processing
- Rationale: Gives founder time to reconsider, negotiate, or convert to credit; client sees "approved" immediately (transparency)
- Trade-offs: Refund not in customer's bank account for 24+ hours (acceptable; better than instant refund regret)
- Research: Founder operational preference; financial safety requirement
- Status: LOCKED IN
- Cost Impact: $0 (operational process, no technical cost)

**D-55-P5-013: Tier 2 Approval Workflow (Asynchronous)**
- Decision: Tier 2 approvals are asynchronous (Rafael reviews on his schedule, not interrupted)
- Rationale: Founder's stated operational preference ("I don't want to sit and stare at a dashboard"); urgent flag escalates if needed
- Trade-offs: Some Lucia decisions are delayed pending Rafael's review (but Tier 1 decisions are immediate)
- Research: Founder operational requirements; operations scaling from 1 person (Rafael) to team
- Status: LOCKED IN
- Cost Impact: $0 (workflow process, no technical cost)

**D-55-P5-014: Monitoring Alerts (SMS for critical, email digest for non-critical)**
- Decision: SMS alerts for critical issues/health check failures; daily email digest for non-critical (background job failures)
- Rationale: Founder not interrupted by non-critical alerts; only emergencies trigger SMS; daily digest provides context
- Trade-offs: Non-critical issues visible only once per day (acceptable; not time-critical)
- Research: Founder operational preference; cost optimization (SMS expensive)
- Status: LOCKED IN

**D-55-P5-015: Testing Framework (Vitest + Playwright)**
- Decision: Unit testing with Vitest, E2E testing with Playwright
- Rationale: Vitest is ESM-native and fast; Playwright is cross-browser; both have strong community support
- Trade-offs: Learning curve; multiple tools (but separated by concern); Playwright slower than integration tests
- Research: Developer experience evaluation; founder requirement for "testing foundation for Phase 7"

- Rationale: Code review enforced; main always deployable; CI/CD catches bugs before production
- Trade-offs: Requires discipline; PR review adds latency; hotfixes need separate process

import { performance } from 'perf_hooks';

interface PerformanceMetric {

const metrics: PerformanceMetric[] = [];

// Middleware to track API endpoint performance
export async function trackApiPerformance(
  req: Request,
  handler: () => Promise<Response>
) {
  const startTime = performance.now();
  const startMark = `${req.method} ${new URL(req.url).pathname}`;
  
  try {
    const response = await handler();
    const duration = performance.now() - startTime;
    
    metrics.push({
      endpoint: new URL(req.url).pathname,
      method: req.method,
      duration_ms: duration,
      status_code: response.status,
      timestamp: new Date().toISOString(),
    });
    
    // Alert if response time exceeds threshold
    if (duration > 500) {
      console.warn(`Slow API: ${req.method} ${new URL(req.url).pathname} took ${duration}ms`);
      // Send to monitoring service (Sentry, Datadog, etc.)
    }
    
    return response;
  } catch (error) {
    const duration = performance.now() - startTime;
    metrics.push({
      endpoint: new URL(req.url).pathname,
      method: req.method,
      duration_ms: duration,
      status_code: 500,
      timestamp: new Date().toISOString(),
      error: String(error),
    });
    throw error;
  }
}

// Aggregate metrics (send to monitoring service every 1 minute)
export function aggregateMetrics() {
  const now = new Date();
  const recentMetrics = metrics.filter(
    (m) => new Date(m.timestamp).getTime() > now.getTime() - 60000
  );
  
  const p95ResponseTime = calculatePercentile(
    recentMetrics.map((m) => m.duration_ms),
    95
  );
  
  const errorRate = (
    (recentMetrics.filter((m) => m.status_code >= 400).length /
      recentMetrics.length) *
    100
  ).toFixed(2);
  
  return {
    total_requests: recentMetrics.length,
    p95_response_time_ms: p95ResponseTime,
    error_rate_percent: errorRate,
    timestamp: new Date().toISOString(),
  };
}
